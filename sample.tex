\begin{frame}[fragile]
  \frametitle{Light Code Listing Style}
  
  \textbf{Using \textbackslash setlststyle for light backgrounds:}
  
  \setlststyle
  \begin{lstlisting}[language=Python]
def hello_world():
    """A simple Python function"""
    message = "Hello, World!"
    print(message)
    return True
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Dark Code Listing Style}
  
  \setbeamercolor{background canvas}{bg=myblack}
  \setbeamercolor*{frametitle}{fg=white}
  \settextcolour{white}
  \setwhitelogofootline{}
  
  \textbf{Using \textbackslash setlststyledark for dark backgrounds:}
  
  \setlststyledark
  \begin{lstlisting}[language=Python]
def hello_world():
    """A simple Python function"""
    message = "Hello, World!"
    print(message)
    return True
  \end{lstlisting}
\end{frame}

\begin{sectionwithpiclargecentral}[nasa-084iI8XTfN0-unsplash.jpg]{Photo by NASA, CC0}{1}
\end{sectionwithpiclargecentral}

\begin{grayframe}
  \frametitle{Inline Text Formatting}
  
  \textbf{Code command:}
  \begin{itemize}
    \item Use \textbackslash code\{text\} for inline code: \code{example code}
    \item Alternative: \code{variable name} for inline formatting
  \end{itemize}
  
  \vspace{1em}
  
  \textbf{Text color control:}
  \begin{itemize}
    \item \textbackslash settextcolour\{white\} - sets all text to white
    \item \textbackslash settextcolour\{myblack\} - sets all text to black
    \item Affects normal text, items, subitems, and blocks
  \end{itemize}
  
  \vspace{1em}
  
  \textbf{Footline customization:}
  \begin{itemize}
    \item \textbackslash setwhitelogofootline - white logo with page numbers
    \item \textbackslash setblacklogofootline - black logo with page numbers
    \item \textbackslash setwhitefootline - no logo, white text
  \end{itemize}
\end{grayframe}
