  \frametitle{Vendor-Managed Runner Pools}
  \begin{columns}[T]
    \column{0.46\textwidth}
    \small
    \settextcolour{myblack}
    \textbf{Challenges:}
    \begin{itemize}
      \item Long lived admin credentials
      \item Runner metadata cannot be enforced
      \item Lack of central administration
      \item Lack of auditability
    \end{itemize}
    
    \column{0.42\textwidth}
    \small
    \settextcolour{white}
    \textbf{Proposed Solution:}
    \begin{itemize}
      \item GHA Runner Token Service (GHARTS)
      \item Use LFID for authentication
      \item No Admin Credentials to vendors
      \item Ephemeral, non reusable credentials
      \item Enforcement of metadata and quota
      \item Centralized administration
      \item Auditability
    \end{itemize}
  \end{columns}
\end{lwhiterblackframeconfig}

  \item GHARTS addresses security and scalability challenges

\begin{blackframe}
  \frametitle{Lessons Learned}
  \small
  \begin{itemize}
    \item \textbf{Clear onboarding:} Streamlined process and transparent decision making
    \item \textbf{Document SLOs:} Define roles and SLOs, include community engagement
    \item \textbf{Reusable Tech Stack:} Share proven configurations and best practices
    \item \textbf{Engage Vendors in WG:} Vendors can and want to contribute to the solution
    \item \textbf{Self Service First:} Distributed engineering effort
    \item \textbf{Empower Maintainers:} Give infra team tools to manage at scale
    \item \textbf{Security:} Consider contributed code and infrastructure as untrusted by default
    \item \textbf{No one-size-fit-all:} Multiple contribution paths reduce attrition
  \end{itemize}
\end{blackframe}

    \item Continue balancing community needs and project sustainability
