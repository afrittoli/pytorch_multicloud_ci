\begin{grayframe}
  \frametitle{Vendor-Managed GitHub Runners}
  \textbf{GitHub Actions Runner Token Service (GHARTS):}
  \begin{itemize}
    \item Secure token distribution for self-hosted runners
    \item Centralized authentication and authorization
    \item Automated runner registration and lifecycle management
    \item Audit logging for all runner operations
  \end{itemize}
  ~\\
  \textbf{Benefits:}
  \begin{itemize}
    \item \textbf{Security:} No long-lived tokens, centralized access control
    \item \textbf{Scalability:} Automated runner provisioning and deprovisioning
    \item \textbf{Visibility:} Centralized monitoring and metrics collection
    \item \textbf{Vendor Integration:} Standardized interface for all vendors
  \end{itemize}
\end{grayframe}
\note[itemize]{
  \item GHARTS solves key security and scalability challenges
  \item Traditional approach uses long-lived tokens (security risk)
  \item GHARTS provides short-lived tokens with proper scoping
  \item Centralized control while vendors manage infrastructure
  \item Automated lifecycle reduces manual overhead
  \item All runner activity logged for audit
  \item Standard interface makes vendor onboarding easier
  \item Enables trust but verify model
}

\begin{grayframe}
  \frametitle{GHARTS Architecture}
  \textbf{Components:}
  \begin{itemize}
    \item \textbf{Token Service:} Issues short-lived runner registration tokens
    \item \textbf{Runner Manager:} Handles runner lifecycle and health checks
    \item \textbf{Metrics Collector:} Aggregates performance and usage data
    \item \textbf{Policy Engine:} Enforces security and access policies
  \end{itemize}
  ~\\
  \textbf{Workflow:}
  \begin{itemize}
    \item Vendor requests token from GHARTS
    \item GHARTS validates vendor credentials and policies
    \item Short-lived token issued for runner registration
    \item Runner registers with GitHub using token
    \item GHARTS monitors runner health and collects metrics
    \item Token expires, runner deregisters automatically
  \end{itemize}
\end{grayframe}
\note[itemize]{
  \item GHARTS is the central control plane
  \item Token service is the security foundation
  \item Short-lived tokens (minutes to hours, not days/months)
  \item Policy engine enforces who can run what workloads
  \item Metrics collector provides visibility
  \item Automated lifecycle reduces operational burden
  \item Vendors can't bypass security controls
  \item System scales to many vendors and runners

\begin{grayframe}
  \frametitle{Vendor-Managed Test Systems}
  \textbf{Accelerator Integration Process:}
  \begin{itemize}
    \item Developed by Accelerator Integration Working Group
    \item Loosely coupled with GitHub Actions
    \item Vendors provide test infrastructure and environments
    \item CI triggers tests via webhooks or API calls
    \item Results reported back to GitHub
  \end{itemize}
  ~\\
  \textbf{Use Cases:}
  \begin{itemize}
    \item Specialized hardware testing (TPUs, custom accelerators)
    \item Long-running performance benchmarks
    \item Platform-specific integration tests
    \item Vendor-specific optimization validation
  \end{itemize}
\end{grayframe}
\note[itemize]{
  \item Different model from GitHub runners
  \item Looser coupling gives vendors more flexibility
  \item Good for specialized hardware not suitable for runners
  \item Vendors control their test environment completely
  \item CI orchestrates but doesn't execute directly
  \item Results still integrated into PR workflow
  \item Lower barrier to entry for some vendors
  \item Accelerator Integration WG created this process
}

\begin{grayframe}

  ~\\
  \textbf{Ongoing Work:}
  \begin{itemize}
    \item \textbf{Standardized Result Format:} Common schema for test results
    \item \textbf{Centralized Dashboard:} Unified view of all test systems
    \item \textbf{Event Relay:} Real-time test status updates
    \item \textbf{Metrics Collection:} Performance and reliability tracking
    \item \textbf{Debugging Tools:} Log aggregation and analysis
  \end{itemize}
\end{grayframe}
\note[itemize]{
  \item Visibility is the main challenge with loose coupling
  \item Hard to see what's happening in vendor systems
  \item Working on standardized reporting
  \item Event relay provides real-time updates
  \item Dashboard gives unified view across all vendors
  \item Metrics help track reliability and performance
  \item Debugging tools help when things go wrong
  \item Goal: maintain flexibility while improving visibility
  \item Balance between vendor autonomy and project needs
}

  \item Security is non-negotiable
